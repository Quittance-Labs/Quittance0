# Quittance frontend — copy to .env.local (local) or set in Vercel Project Settings

# Public API base URL (must include /api)
# Local:  http://localhost:3001/api
# Prod:   https://YOUR-API-HOST/api
NEXT_PUBLIC_API_URL=http://localhost:3001/api

# Stellar network: TESTNET or PUBLIC
NEXT_PUBLIC_STELLAR_NETWORK=TESTNET
NEXT_PUBLIC_HORIZON_URL=https://horizon-testnet.stellar.org

# Public site URL (no trailing slash)
# Local:  http://localhost:3000
# Prod:   https://YOUR-FRONTEND.vercel.app
NEXT_PUBLIC_APP_NAME=Quittance
NEXT_PUBLIC_APP_URL=http://localhost:3000

# Always false for real backend (mock is local UI-only)
NEXT_PUBLIC_USE_MOCK=false
